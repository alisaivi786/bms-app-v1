# Progress Tracking

Learn how to track upload and download progress for your API requests.

## Overview

The `onProgress` callback allows you to track the progress of both uploads and downloads. This works with both Axios and
Fetch implementations.

### Progress Event Structure

```typescript
interface ProgressEvent {
	loaded: number; // Bytes transferred so far
	total: number; // Total bytes to transfer
	percentage: number; // Progress as percentage (0-100)
}
```

### Key Features

- ✅ Works with both Axios and Fetch
- ✅ Supports uploads and downloads
- ✅ Provides byte count and percentage
- ✅ Real-time updates during transfer

## Basic Usage

### Simple Progress Tracking

```typescript
import { useLazyApi } from '@devkit/api-client';
import { uploadFileApi } from '../api/files';

function FileUploader() {
	const { callApi, isLoading } = useLazyApi(uploadFileApi);
	const [progress, setProgress] = React.useState(0);

	const handleUpload = async (file: File) => {
		await callApi(
			{ file },
			{
				onProgress: (event) => {
					setProgress(event.percentage);
				},
			}
		);
	};

	return (
		<div>
			<input
				type="file"
				onChange={(e) => {
					const file = e.target.files?.[0];
					if (file) handleUpload(file);
				}}
			/>
			{isLoading && (
				<div>
					<progress value={progress} max={100} />
					<span>{progress}%</span>
				</div>
			)}
		</div>
	);
}
```

## Upload Progress

### File Upload with Progress Bar

```typescript
function AdvancedFileUploader() {
	const { callApi, isLoading, result } = useLazyApi(uploadFileApi);
	const [progress, setProgress] = React.useState(0);
	const [uploadSpeed, setUploadSpeed] = React.useState(0);
	const [file, setFile] = React.useState<File | null>(null);

	const startTimeRef = React.useRef<number>(0);
	const lastLoadedRef = React.useRef<number>(0);

	const handleUpload = async () => {
		if (!file) return;

		startTimeRef.current = Date.now();
		lastLoadedRef.current = 0;

		await callApi(
			{ file },
			{
				onProgress: (event) => {
					setProgress(event.percentage);

					// Calculate upload speed
					const elapsed = (Date.now() - startTimeRef.current) / 1000; // seconds
					const bytesPerSecond = event.loaded / elapsed;
					setUploadSpeed(bytesPerSecond);
				},
			}
		);
	};

	const formatBytes = (bytes: number) => {
		if (bytes === 0) return '0 Bytes';
		const k = 1024;
		const sizes = ['Bytes', 'KB', 'MB', 'GB'];
		const i = Math.floor(Math.log(bytes) / Math.log(k));
		return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
	};

	return (
		<div className="uploader">
			<input type="file" onChange={(e) => setFile(e.target.files?.[0] || null)} />
			<button onClick={handleUpload} disabled={!file || isLoading}>
				{isLoading ? 'Uploading...' : 'Upload'}
			</button>

			{isLoading && (
				<div className="progress-info">
					<div className="progress-bar">
						<div className="progress-fill" style={{ width: `${progress}%` }} />
					</div>
					<div className="stats">
						<span>{progress.toFixed(1)}%</span>
						<span>{formatBytes(uploadSpeed)}/s</span>
					</div>
				</div>
			)}

			{result && (
				<div className="success">
					File uploaded successfully!
					<a href={result.url}>View file</a>
				</div>
			)}
		</div>
	);
}
```

### Multiple File Upload

```typescript
interface UploadItem {
	file: File;
	progress: number;
	status: 'pending' | 'uploading' | 'completed' | 'error';
	error?: string;
}

function MultiFileUploader() {
	const { callApi } = useLazyApi(uploadFileApi);
	const [uploads, setUploads] = React.useState<UploadItem[]>([]);

	const handleFiles = (files: FileList) => {
		const newUploads: UploadItem[] = Array.from(files).map((file) => ({
			file,
			progress: 0,
			status: 'pending',
		}));
		setUploads((prev) => [...prev, ...newUploads]);

		// Start uploading
		newUploads.forEach((upload, index) => {
			uploadFile(upload.file, uploads.length + index);
		});
	};

	const uploadFile = async (file: File, index: number) => {
		setUploads((prev) => prev.map((item, i) => (i === index ? { ...item, status: 'uploading' } : item)));

		try {
			await callApi(
				{ file },
				{
					onProgress: (event) => {
						setUploads((prev) => prev.map((item, i) => (i === index ? { ...item, progress: event.percentage } : item)));
					},
				}
			);

			setUploads((prev) => prev.map((item, i) => (i === index ? { ...item, status: 'completed' } : item)));
		} catch (error) {
			setUploads((prev) =>
				prev.map((item, i) => (i === index ? { ...item, status: 'error', error: error.message } : item))
			);
		}
	};

	return (
		<div>
			<input
				type="file"
				multiple
				onChange={(e) => {
					if (e.target.files) {
						handleFiles(e.target.files);
					}
				}}
			/>

			<div className="upload-list">
				{uploads.map((upload, index) => (
					<div key={index} className="upload-item">
						<div className="file-name">{upload.file.name}</div>
						<div className="progress-bar">
							<div className="progress-fill" style={{ width: `${upload.progress}%` }} />
						</div>
						<div className="status">{upload.status}</div>
						{upload.error && <div className="error">{upload.error}</div>}
					</div>
				))}
			</div>
		</div>
	);
}
```

### Form Data Upload

```typescript
function ProfilePictureUploader() {
	const { callApi, isLoading } = useLazyApi(updateProfileApi);
	const [progress, setProgress] = React.useState(0);

	const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
		e.preventDefault();
		const formData = new FormData(e.currentTarget);

		await callApi(
			{
				name: formData.get('name') as string,
				bio: formData.get('bio') as string,
				avatar: formData.get('avatar') as File,
			},
			{
				onProgress: (event) => {
					setProgress(event.percentage);
				},
			}
		);
	};

	return (
		<form onSubmit={handleSubmit}>
			<input name="name" placeholder="Name" />
			<textarea name="bio" placeholder="Bio" />
			<input type="file" name="avatar" accept="image/*" />

			<button type="submit" disabled={isLoading}>
				{isLoading ? `Uploading... ${progress}%` : 'Save Profile'}
			</button>

			{isLoading && <progress value={progress} max={100} />}
		</form>
	);
}
```

## Download Progress

### File Download with Progress

```typescript
function FileDownloader({ fileId }: { fileId: string }) {
	const { callApi, isLoading } = useLazyApi(downloadFileApi);
	const [progress, setProgress] = React.useState(0);
	const [fileUrl, setFileUrl] = React.useState<string>('');

	const downloadFile = async () => {
		const result = await callApi(
			{ parameters: { fileId } },
			{
				onProgress: (event) => {
					setProgress(event.percentage);
				},
			}
		);

		// Create blob URL for download
		const blob = new Blob([result.data]);
		const url = URL.createObjectURL(blob);
		setFileUrl(url);
	};

	React.useEffect(() => {
		return () => {
			if (fileUrl) {
				URL.revokeObjectURL(fileUrl);
			}
		};
	}, [fileUrl]);

	return (
		<div>
			<button onClick={downloadFile} disabled={isLoading}>
				Download File
			</button>

			{isLoading && (
				<div>
					<progress value={progress} max={100} />
					<span>Downloading... {progress}%</span>
				</div>
			)}

			{fileUrl && (
				<a href={fileUrl} download="file.zip">
					Click to save file
				</a>
			)}
		</div>
	);
}
```

### Large Data Fetch with Progress

```typescript
function DataExporter() {
	const { callApi, isLoading } = useLazyApi(exportDataApi);
	const [progress, setProgress] = React.useState(0);
	const [estimatedTime, setEstimatedTime] = React.useState(0);

	const startTimeRef = React.useRef<number>(0);

	const exportData = async () => {
		startTimeRef.current = Date.now();

		await callApi(
			{ format: 'csv' },
			{
				onProgress: (event) => {
					setProgress(event.percentage);

					// Calculate estimated time remaining
					const elapsed = Date.now() - startTimeRef.current;
					const rate = event.loaded / elapsed;
					const remaining = event.total - event.loaded;
					const estimate = remaining / rate / 1000; // seconds

					setEstimatedTime(estimate);
				},
			}
		);
	};

	return (
		<div>
			<button onClick={exportData} disabled={isLoading}>
				Export Data
			</button>

			{isLoading && (
				<div className="progress-container">
					<div className="progress-bar">
						<div className="fill" style={{ width: `${progress}%` }} />
					</div>
					<div className="info">
						<span>{progress.toFixed(1)}%</span>
						<span>{estimatedTime > 0 && `~${Math.ceil(estimatedTime)}s remaining`}</span>
					</div>
				</div>
			)}
		</div>
	);
}
```

## Advanced Patterns

### Combined Upload/Download Progress

```typescript
function FileConverter() {
	const { callApi: upload } = useLazyApi(uploadForConversionApi);
	const { callApi: download } = useLazyApi(downloadConvertedApi);

	const [uploadProgress, setUploadProgress] = React.useState(0);
	const [downloadProgress, setDownloadProgress] = React.useState(0);
	const [stage, setStage] = React.useState<'idle' | 'uploading' | 'processing' | 'downloading'>('idle');

	const convertFile = async (file: File) => {
		// Stage 1: Upload
		setStage('uploading');
		const uploadResult = await upload(
			{ file },
			{
				onProgress: (event) => {
					setUploadProgress(event.percentage);
				},
			}
		);

		// Stage 2: Processing (no progress)
		setStage('processing');

		// Stage 3: Download
		setStage('downloading');
		await download(
			{ parameters: { jobId: uploadResult.jobId } },
			{
				onProgress: (event) => {
					setDownloadProgress(event.percentage);
				},
			}
		);

		setStage('idle');
	};

	return (
		<div>
			<input
				type="file"
				onChange={(e) => {
					const file = e.target.files?.[0];
					if (file) convertFile(file);
				}}
			/>

			{stage !== 'idle' && (
				<div className="conversion-progress">
					<div className="stage">
						{stage === 'uploading' && (
							<>
								<span>Uploading...</span>
								<progress value={uploadProgress} max={100} />
							</>
						)}
						{stage === 'processing' && (
							<>
								<span>Processing...</span>
								<div className="spinner" />
							</>
						)}
						{stage === 'downloading' && (
							<>
								<span>Downloading...</span>
								<progress value={downloadProgress} max={100} />
							</>
						)}
					</div>
				</div>
			)}
		</div>
	);
}
```

### Progress with Retry

```typescript
function ResilientUploader() {
	const { callApi } = useLazyApi(uploadFileApi);
	const [progress, setProgress] = React.useState(0);
	const [attempt, setAttempt] = React.useState(0);
	const maxAttempts = 3;

	const uploadWithRetry = async (file: File) => {
		for (let i = 0; i < maxAttempts; i++) {
			setAttempt(i + 1);
			setProgress(0);

			try {
				await callApi(
					{ file },
					{
						onProgress: (event) => {
							setProgress(event.percentage);
						},
					}
				);

				// Success!
				return;
			} catch (error) {
				if (i === maxAttempts - 1) {
					throw error; // Last attempt failed
				}

				// Wait before retry
				await new Promise((resolve) => setTimeout(resolve, 1000 * (i + 1)));
			}
		}
	};

	return (
		<div>
			{attempt > 0 && (
				<div>
					Attempt {attempt}/{maxAttempts}
					<progress value={progress} max={100} />
				</div>
			)}
		</div>
	);
}
```

### Progress Persistence

```typescript
function PersistentUploader() {
	const { callApi } = useLazyApi(uploadFileApi);
	const [progress, setProgress] = React.useState(() => {
		// Restore from localStorage
		const saved = localStorage.getItem('upload-progress');
		return saved ? parseInt(saved, 10) : 0;
	});

	const upload = async (file: File) => {
		await callApi(
			{ file },
			{
				onProgress: (event) => {
					setProgress(event.percentage);
					// Save to localStorage
					localStorage.setItem('upload-progress', event.percentage.toString());
				},
			}
		);

		// Clear on success
		localStorage.removeItem('upload-progress');
	};

	return (
		<div>
			<progress value={progress} max={100} />
			{progress > 0 && progress < 100 && <div>Resuming from {progress}%</div>}
		</div>
	);
}
```

## Best Practices

### 1. Show Visual Feedback

```typescript
// ✅ Good: Clear visual progress
<div className="progress-wrapper">
  <progress value={progress} max={100} />
  <span>{progress}%</span>
</div>

// ❌ Bad: No visual feedback
<div>Uploading...</div>
```

### 2. Handle Edge Cases

```typescript
// ✅ Good: Handle missing total
onProgress: (event) => {
	if (event.total === 0) {
		// Show indeterminate progress
		setIsUploading(true);
	} else {
		setProgress(event.percentage);
	}
};

// ❌ Bad: Assume total is always available
onProgress: (event) => {
	setProgress(event.percentage);
};
```

### 3. Provide Additional Information

```typescript
// ✅ Good: Show speed and time remaining
<div>
	<progress value={progress} max={100} />
	<div>
		{formatBytes(loaded)} / {formatBytes(total)}
		<span>{formatBytes(speed)}/s</span>
		<span>~{formatTime(remaining)} remaining</span>
	</div>
</div>
```

### 4. Allow Cancellation

```typescript
// ✅ Good: Provide cancel button
const abortControllerRef = React.useRef<AbortController>();

const upload = async () => {
	abortControllerRef.current = new AbortController();
	// Pass signal to API
};

<button onClick={() => abortControllerRef.current?.abort()}>Cancel</button>;
```

### 5. Throttle Updates

```typescript
// ✅ Good: Throttle for better performance
const lastUpdateRef = React.useRef(0);

onProgress: (event) => {
	const now = Date.now();
	if (now - lastUpdateRef.current > 100) {
		// Update max 10x/second
		setProgress(event.percentage);
		lastUpdateRef.current = now;
	}
};

// ❌ Bad: Update on every event (can cause performance issues)
onProgress: (event) => {
	setProgress(event.percentage);
};
```

### 6. Reset State

```typescript
// ✅ Good: Reset progress between uploads
const startUpload = async (file: File) => {
  setProgress(0);
  setError(null);
  await upload(file);
};

// ❌ Bad: Old progress remains visible
const startUpload = async (file: File) => {
  await upload(file);
};
```

### Circular Progress

```typescript
function CircularProgress({ progress }: { progress: number }) {
	const radius = 40;
	const circumference = 2 * Math.PI * radius;
	const offset = circumference - (progress / 100) * circumference;

	return (
		<svg width="100" height="100">
			<circle cx="50" cy="50" r={radius} fill="none" stroke="#e0e0e0" strokeWidth="8" />
			<circle
				cx="50"
				cy="50"
				r={radius}
				fill="none"
				stroke="#4caf50"
				strokeWidth="8"
				strokeDasharray={circumference}
				strokeDashoffset={offset}
				transform="rotate(-90 50 50)"
			/>
			<text x="50" y="55" textAnchor="middle" fontSize="16">
				{progress}%
			</text>
		</svg>
	);
}
```
